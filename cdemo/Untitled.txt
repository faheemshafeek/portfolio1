class DoctorRegView(View):
    def get(self,request):
        form=DoctorForm()
        return render(request,"addoc.html",{"form":form})
    def post(self,request):
        fdata=DoctorForm(request.POST)
        if fdata.is_valid():
           fname=request.POST.get("fname")
           return HttpResponse(fname)
        else:
            return render(request,"addoc.html",{"form":fdata})
